# 带符号的大数相加

```java
public class Main {
    static class ListNode {
        char val;
        ListNode next;

        ListNode(char val) {
            this.val = val;
        }
    }

    public static ListNode addTwoList(ListNode l1, ListNode l2) {
        ListNode dummy = new ListNode('0'), cur = dummy;
        boolean neg, neg1 = false, neg2 = false;

        if (l1.val == '-') {
            neg1 = true;
            l1 = l1.next;
        }
        if (l2.val == '-') {
            neg2 = true;
            l2 = l2.next;
        }

        if (l1.val == '0' && l2.val == '0') return l1;

        boolean less = compare(l1, l2) < 0;
        ListNode c1 = reverse(l1), c2 = reverse(l2);

        int carry = 0;
        if (neg1 != neg2) {
            // 保证 s1 的数字大于等于 s2 的数字
            if (less) {
                ListNode tmp = c1;
                c1 = c2;
                c2 = tmp;
                neg = neg2;
            } else {
                neg = neg1;
            }

            while (c1 != null || c2 != null) {
                if (c1 != null) {
                    carry += (c1.val - '0');
                    c1 = c1.next;
                }
                if (c2 != null) {
                    carry -= (c2.val - '0');
                    c2 = c2.next;
                }

                if (c1 == null && c2 == null && carry == 0) break;

                if (carry < 0) {
                    cur.next = new ListNode((char) (carry + 10 + '0'));
                    carry = -1;
                } else {
                    cur.next = new ListNode((char) (carry + '0'));
                    carry = 0;
                }
                cur = cur.next;
            }
        } else {
            neg = neg1;
            while (c1 != null || c2 != null || carry != 0) {
                if (c1 != null) {
                    carry += (c1.val - '0');
                    c1 = c1.next;
                }
                if (c2 != null) {
                    carry += (c2.val - '0');
                    c2 = c2.next;
                }
                cur.next = new ListNode((char) (carry % 10 + '0'));
                cur = cur.next;
                carry /= 10;
            }
        }
        if (neg) cur.next = new ListNode('-');
        return reverse(dummy.next);
    }

    private static ListNode reverse(ListNode head) {
        ListNode pre = null, cur = null;
        while (head != null) {
            cur = head;
            head = head.next;
            cur.next = pre;
            pre = cur;
        }
        return cur;
    }

    private static int len(ListNode l1) {
        int len = 0;
        while (l1 != null) {
            len++;
            l1 = l1.next;
        }
        return len;
    }

    private static int compare(ListNode l1, ListNode l2) {
        int len1 = len(l1), len2 = len(l2);
        if (len1 == len2) {
            while (l1 != null) {
                if (l1.val != l2.val) return l1.val - l2.val;
                else {
                    l1 = l1.next;
                    l2 = l2.next;
                }
            }
            return 0;
        } else {
            return len1 - len2;
        }
    }

    public static ListNode create(String str) {
        ListNode dummy = new ListNode('0'), cur = dummy;
        for (char ch : str.toCharArray()) {
            cur.next = new ListNode(ch);
            cur = cur.next;
        }
        return dummy.next;
    }

    public static void print(ListNode head) {
        ListNode cur = head;
        StringBuilder builder = new StringBuilder();
        while (cur != null) {
            builder.append(cur.val);
            cur = cur.next;
        }
        System.out.println(builder.toString());
    }

    public static void main(String[] args) {
        ListNode l1 = create("-0");
        ListNode l2 = create("0");
        print(addTwoList(l1, l2));
    }
}
```
