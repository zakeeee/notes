# 最短路径

```java
public class Main {
    public static final int INF = 10000;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N, P, C, total;
        int[][] weight;
        int[] distTo;
        while (sc.hasNext()) {
            N = sc.nextInt(); P = sc.nextInt(); C = sc.nextInt();
            // 构建邻接矩阵
            weight = new int[N][N]; distTo = new int[N];
            for (int i = 0; i < N; i++) {
                for (int j = 0; j < N; j++) {
                    weight[i][j] = INF;
                }
            }
            for (int i = 0; i < P; i++) {
                int src = sc.nextInt(), tgt = sc.nextInt(), time = sc.nextInt();
                weight[src][tgt] = time;
            }

            // 计算从 0 到每个点的最短耗时
            dijkstra(weight, distTo, 0, N);

            total = 0;
            for (int i = 0; i < C; i++) total += distTo[sc.nextInt()];

            System.out.println(Arrays.toString(distTo));
            System.out.println(total);
        }
    }

    private static void dijkstra(int[][] weight, int[] distTo, int src, int N) {
        for (int i = 0; i < N; i++) distTo[i] = INF;
        distTo[src] = 0;

        PriorityQueue<Integer> queue = new PriorityQueue<>(Comparator.comparingInt(a -> distTo[a]));
        Set<Integer> set = new HashSet<>();  // 保存添加到队列过的节点索引
        queue.add(src);
        set.add(src);

        while (!queue.isEmpty()) {
            src = queue.poll();
            for (int tgt = 0; tgt < N; tgt++) {
                // 跳过自己或者不可达的目标
                if (src == tgt || weight[src][tgt] == INF) continue;

                if (distTo[tgt] > distTo[src] + weight[src][tgt]) {
                    distTo[tgt] = distTo[src] + weight[src][tgt];

                    if (set.contains(tgt)) {
                        queue.remove(tgt);
                    } else {
                        set.add(tgt);
                    }
                    queue.add(tgt);
                }
            }
        }
    }
}
```
