# 并查集

```java
public class UnionFind {
    private int[] roots;
    private int cnt;

    public UnionFind(int n) {
        this.roots = new int[n];
        this.cnt = n;
        for (int i = 0; i < n; i++) {
            roots[i] = i;
        }
    }

    public void union(int a, int b) {
        int p = find(a), q = find(b);
        if (p != q) {
            roots[p] = q;
            cnt--;
        }
    }

    public int find(int a) {
        if (roots[a] != a) roots[a] = find(roots[a]);
        return roots[a];
    }
```
